1 /++
2 A sum type for modern D.
3 
4 This module provides [SumType], an alternative to `std.variant.Algebraic` with
5 [match|improved pattern-matching], full attribute correctness (`pure`, `@safe`,
6 `@nogc`, and `nothrow` are inferred whenever possible), and no dependency on
7 runtime type information (`TypeInfo`).

14 /// $(H3 Basic usage)
15 unittest {
16 	import std.math: approxEqual;
17 
18 	struct Fahrenheit { double degrees; }
19 	struct Celsius { double degrees; }
20 	struct Kelvin { double degrees; }
21 
22 	alias Temperature = SumType!(Fahrenheit, Celsius, Kelvin);
23 
24 	pure @safe @nogc nothrow
25 	Fahrenheit toFahrenheit(Temperature t)
26 	{
27 		return Fahrenheit(
28 			t.match!(
29 				(Fahrenheit f) => f.degrees,
30 				(Celsius c) => c.degrees * 9.0/5 + 32,
31 				(Kelvin k) => k.degrees * 9.0/5 - 459.4
32 			)
33 		);
34 	}
35 
36 	Temperature t1 = Fahrenheit(98.6);
37 	Temperature t2 = Celsius(100);
38 	Temperature t3 = Kelvin(273);
39 
40 	assert(toFahrenheit(t1).degrees.approxEqual(98.6));
41 	assert(toFahrenheit(t2).degrees.approxEqual(212));
42 	assert(toFahrenheit(t3).degrees.approxEqual(32));
43 }

45 /** $(H3 Structural matching)
46  *
47  * In the `length` and `horiz` functions below, the handlers for `match` do not
48  * specify the types of their arguments. Instead, matching is done based on the
49  * type's structure: any type with `x` and `y` properties will be matched by the
50  * `rect` handlers, and any type with `r` and `theta` properties will be matched
51  * by the `polar` handlers.
52  */
53 unittest {
54 	import std.math: approxEqual, cos, PI, sqrt;
55 
56 	struct Rectangular { double x, y; }
57 	struct Polar { double r, theta; }
58 	alias Vector = SumType!(Rectangular, Polar);
59 
60 	pure @safe @nogc nothrow
61 	double length(Vector v)
62 	{
63 		return v.match!(
64 			rect => sqrt(rect.x^^2 + rect.y^^2),
65 			polar => polar.r
66 		);
67 	}
68 
69 	pure @safe @nogc nothrow
70 	double horiz(Vector v)
71 	{
72 		return v.match!(
73 			rect => rect.x,
74 			polar => polar.r * cos(polar.theta)
75 		);
76 	}
77 
78 	Vector u = Rectangular(1, 1);
79 	Vector v = Polar(1, PI/4);
80 
81 	assert(length(u).approxEqual(sqrt(2.0)));
82 	assert(length(v).approxEqual(1));
83 	assert(horiz(u).approxEqual(1));
84 	assert(horiz(v).approxEqual(sqrt(0.5)));
85 }

87 /** $(H3 Arithmetic expression evaluator)
88  *
89  * This example makes use of the special placeholder type `This` to define a
90  * [https://en.wikipedia.org/wiki/Recursive_data_type|recursive data type]: an
91  * [https://en.wikipedia.org/wiki/Abstract_syntax_tree|abstract syntax tree] for
92  * representing simple arithmetic expressions.
93  */
94 unittest {
95 	import std.functional: partial;
96 	import std.traits: EnumMembers;
97 	import std.typecons: Tuple;
98 
99 	enum Op : string
100 	{
101 		Plus  = "+",
102 		Minus = "-",
103 		Times = "*",
104 		Div   = "/"
105 	}
106 
107 	// An expression is either
108 	//  - a number,
109 	//  - a variable, or
110 	//  - a binary operation combining two sub-expressions.
111 	alias Expr = SumType!(
112 		double,
113 		string,
114 		Tuple!(Op, "op", This*, "lhs", This*, "rhs")
115 	);
116 
117 	// Shorthand for the Tuple type above
118 	alias BinOp = Expr.Types[2];
119 
120 	// Factory function for number expressions
121 	pure @safe
122 	Expr* num(double value)
123 	{
124 		return new Expr(value);
125 	}
126 
127 	// Factory function for variable expressions
128 	pure @safe
129 	Expr* var(string name)
130 	{
131 		return new Expr(name);
132 	}
133 
134 	// Factory function for binary operation expressions
135 	pure @safe
136 	Expr* binOp(Op op, Expr* lhs, Expr* rhs)
137 	{
138 		return new Expr(BinOp(op, lhs, rhs));
139 	}
140 
141 	// Convenience wrappers for creating BinOp expressions
142 	alias sum  = partial!(binOp, Op.Plus);
143 	alias diff = partial!(binOp, Op.Minus);
144 	alias prod = partial!(binOp, Op.Times);
145 	alias quot = partial!(binOp, Op.Div);
146 
147 	// Evaluate expr, looking up variables in env
148 	pure @safe nothrow
149 	double eval(Expr expr, double[string] env)
150 	{
151 		return expr.match!(
152 			(double num) => num,
153 			(string var) => env[var],
154 			(BinOp bop) {
155 				double lhs = eval(*bop.lhs, env);
156 				double rhs = eval(*bop.rhs, env);
157 				final switch(bop.op) {
158 					static foreach(op; EnumMembers!Op) {
159 						case op:
160 							return mixin("lhs" ~ op ~ "rhs");
161 					}
162 				}
163 			}
164 		);
165 	}
166 
167 	// Return a "pretty-printed" representation of expr
168 	@safe
169 	string pprint(Expr expr)
170 	{
171 		import std.format;
172 
173 		return expr.match!(
174 			(double num) => "%g".format(num),
175 			(string var) => var,
176 			(BinOp bop) => "(%s %s %s)".format(
177 				pprint(*bop.lhs),
178 				bop.op,
179 				pprint(*bop.rhs)
180 			)
181 		);
182 	}
183 
184 	Expr* myExpr = sum(var("a"), prod(num(2), var("b")));
185 	double[string] myEnv = ["a":3, "b":4, "c":7];
186 
187 	assert(eval(*myExpr, myEnv) == 11);
188 	assert(pprint(*myExpr) == "(a + (2 * b))");
189 }

193 /**
194  * A tagged union that can hold a single value from any of a specified set of
195  * types
196  *
197  * The stored value can be operated on using [match|pattern matching].
198  *
199  * The special type `This` can be used as a placeholder to create
200  * self-referential types, just like with `Algebraic`. See the
201  * [sumtype#arithmetic-expression-evaluator|"Arithmetic expression evaluator" example] for
202  * usage.
203  *
204  * A `SumType` is initialized by default with the `.init` property of its first
205  * member type, just like a regular union.
206  *
207  * See_Also: `std.variant.Algebraic`
208  */
209 struct SumType(TypeArgs...)
210 {
211 	import std.meta: AliasSeq;
212 	import std.typecons: ReplaceType;
213 
214 	/// The types a `SumType` can hold
215 	alias Types = AliasSeq!(ReplaceType!(This, typeof(this), TypeArgs));
216 
217 private:
218 
219 	int tag;
220 
221 	union Storage
222 	{
223 		Types values;
224 	}
225 
226 	Storage storage;
227 
228 public:
229 
230 	static foreach (i, T; Types) {
231 		/// Constructs a `SumType` holding a specific value
232 		this(T val)
233 		{
234 			tag = i;
235 			storage.values[i] = val;
236 		}
237 	}
238 
239 	import std.traits: isAssignable;
240 
241 	static foreach (i, T; Types) {
242 		static if (isAssignable!T) {
243 			/// Assigns a value to a `SumType`
244 			void opAssign(T rhs)
245 			{
246 				import std.algorithm.mutation: moveEmplace;
247 
248 				this.match!((ref value) {
249 					static if (hasElaborateDestructor!(typeof(value))) {
250 						destroy(value);
251 					}
252 				});
253 
254 				tag = i;
255 				() @trusted { moveEmplace(rhs, storage.values[i]); }();
256 			}
257 		}
258 	}
259 
260 	/// Returns a string representation of the currently-held value
261 	string toString()
262 	{
263 		import std.conv: to;
264 
265 		return this.match!(value => value.to!string);
266 	}
267 
268 	import std.meta: anySatisfy;
269 	import std.traits: hasElaborateCopyConstructor, hasElaborateDestructor;
270 
271 	static if (anySatisfy!(hasElaborateDestructor, Types)) {
272 		~this()
273 		{
274 			this.match!((ref value) {
275 				static if (hasElaborateDestructor!(typeof(value))) {
276 					destroy(value);
277 				}
278 			});
279 		}
280 	}
281 
282 	static if (anySatisfy!(hasElaborateCopyConstructor, Types)) {
283 		this(this)
284 		{
285 			this.match!((ref value) {
286 				static if (hasElaborateCopyConstructor!(typeof(value))) {
287 					value.__xpostblit;
288 				}
289 			});
290 		}
291 	}
292 }

500 /**
501  * Calls a type-appropriate function with the value held in a [SumType].
502  *
503  * For each possible type the [SumType] can hold, the given handlers are
504  * checked, in order, to see whether they accept a single argument of that type.
505  * The first one that does is chosen as the match for that type.
506  *
507  * Implicit conversions are not taken into account, except between
508  * differently-qualified versions of the same type. For example, a handler that
509  * accepts a `long` will not match the type `int`, but a handler that accepts a
510  * `const(int)[]` will match the type `immutable(int)[]`.
511  *
512  * Every type must have a matching handler. This is enforced at compile-time.
513  *
514  * Handlers may be functions, delegates, or objects with opCall overloads.
515  * Templated handlers are also accepted, and will match any type for which they
516  * can be implicitly instantiated. See [sumtype#structural-matching|"Structural matching"] for
517  * an example of templated handler usage.
518  *
519  * Returns:
520  *   The value returned from the handler that matches the currently-held type.
521  *
522  * See_Also: `std.variant.visit`
523  */
524 template match(handlers...)
525 {
526 	import std.typecons: Yes;
527 
528 	/**
529 	 * The actual `match` function.
530 	 *
531 	 * Params:
532 	 *   self = A [SumType] object
533 	 */
534 	auto match(Self)(auto ref Self self)
535 		if (is(Self : SumType!TypeArgs, TypeArgs...))
536 	{
537 		return self.matchImpl!(Yes.exhaustive, handlers);
538 	}
539 }
540 
541 /**
542  * Attempts to call a type-appropriate function with the value held in a
543  * [SumType], and throws on failure.
544  *
545  * Matches are chosen using the same rules as [match], but are not required to
546  * be exhaustive—in other words, a type is allowed to have no matching handler.
547  * If a type without a handler is encountered at runtime, a [MatchException]
548  * is thrown.
549  *
550  * Returns:
551  *   The value returned from the handler that matches the currently-held type,
552  *   if a handler was given for that type.
553  *
554  * Throws:
555  *   [MatchException], if the currently-held type has no matching handler.
556  *
557  * See_Also: `std.variant.tryVisit`
558  */
559 template tryMatch(handlers...)
560 {
561 	import std.typecons: No;
562 
563 	/**
564 	 * The actual `tryMatch` function.
565 	 *
566 	 * Params:
567 	 *   self = A [SumType] object
568 	 */
569 	auto tryMatch(Self)(auto ref Self self)
570 		if (is(Self : SumType!TypeArgs, TypeArgs...))
571 	{
572 		return self.matchImpl!(No.exhaustive, handlers);
573 	}
574 }
575 
576 /// Thrown by [tryMatch] when an unhandled type is encountered.
577 class MatchException : Exception
578 {
579 	this(string msg, string file = __FILE__, size_t line = __LINE__)
580 	{
581 		super(msg, file, line);
582 	}
583 }
584 
585 import std.typecons: Flag;
586 
587 private template matchImpl(Flag!"exhaustive" exhaustive, handlers...)
588 {
589 	auto matchImpl(Self)(auto ref Self self)
590 		if (is(Self : SumType!TypeArgs, TypeArgs...))
591 	{
592 		alias Types = self.Types;
593 
594 		pure static int[Types.length] getHandlerIndices()
595 		{
596 			import std.traits: hasMember, isCallable, isSomeFunction, Parameters;
597 
598 			// immutable recursively overrides all other qualifiers, so the
599 			// right-hand side is true if and only if the two types are the
600 			// same up to qualifiers (i.e., they have the same structure).
601 			enum sameUpToQuals(T, U) = is(immutable(T) == immutable(U));
602 
603 			int[Types.length] indices;
604 			indices[] = -1;
605 
606 			void setHandlerIndex(int tid, int hid)
607 			{
608 				if (indices[tid] == -1) {
609 					indices[tid] = hid;
610 				}
611 			}
612 
613 			static foreach (tid, T; Types) {
614 				static foreach (hid, handler; handlers) {
615 					static if (is(typeof(handler(self.storage.values[tid])))) {
616 						// Regular handlers
617 						static if (isCallable!handler) {
618 							// Functions and delegates
619 							static if (isSomeFunction!handler) {
620 								static if (sameUpToQuals!(T, Parameters!handler[0])) {
621 									setHandlerIndex(tid, hid);
622 								}
623 							// Objects with overloaded opCall
624 							} else static if (hasMember!(typeof(handler), "opCall")) {
625 								static foreach (overload; __traits(getOverloads, typeof(handler), "opCall")) {
626 									static if (sameUpToQuals!(T, Parameters!overload[0])) {
627 										setHandlerIndex(tid, hid);
628 									}
629 								}
630 							}
631 						// Generic handlers
632 						} else {
633 							setHandlerIndex(tid, hid);
634 						}
635 					}
636 				}
637 			}
638 			return indices;
639 		}
640 
641 		enum handlerIndices = getHandlerIndices;
642 
643 		import std.algorithm.searching: canFind;
644 
645 		static foreach (hid, handler; handlers) {
646 			static assert(handlerIndices[].canFind(hid),
647 				"handler `" ~ handler.stringof ~ "` " ~
648 				"of type `" ~ typeof(handler).stringof ~ "` " ~
649 				"never matches"
650 			);
651 		}
652 
653 		final switch (self.tag) {
654 			static foreach (tid, T; Types) {
655 				case tid:
656 					static if (handlerIndices[tid] != -1) {
657 						return handlers[handlerIndices[tid]](self.storage.values[tid]);
658 					} else {
659 						static if(exhaustive) {
660 							static assert(0,
661 								"No matching handler for type `" ~ T.stringof ~ "`");
662 						} else {
663 							throw new MatchException(
664 								"No matching handler for type `" ~ T.stringof ~ "`");
665 						}
666 					}
667 			}
668 		}
669 		assert(false); // unreached
670 	}
671 }
